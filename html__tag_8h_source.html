181 template <class T>
182 inline T &get()
183 {
184 T* searched_type = nullptr;
185 std::string error_msg;
186
187 if constexpr (std::derived_from<T, tags::itag>)
188 {
189 auto it = std::ranges::find_if(subelements_, search_by_type<T>);
190
191 if(it != subelements_.end())
192 searched_type = dynamic_cast<T*>(it->get());
193 else
194 error_msg = std::format("Tag <{}> was not found in <{}>", T().tag_name(), this->tag_name());
195 }
196 else
197 {
198 auto it = std::ranges::find_if(attributes_, search_by_type<T>);
199
200 if(it != attributes_.end())
201 searched_type = dynamic_cast<T*>(it->get());
202 else
203 error_msg = std::format("Attribute \"{}\" was not found in <{}>", htmcxx::utils::get_class_name(T("")), this->tag_name());
204 }
205
206 return searched_type? *searched_type : throw std::out_of_range(error_msg);
207 }
